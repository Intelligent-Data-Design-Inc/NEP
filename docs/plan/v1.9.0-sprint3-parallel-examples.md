## Tasks

### 1. Create `examples/parallelIO/square16_par.c`

```c
/* square16_par.c - Parallel I/O hello: print MPI rank and size. */
/* Edward Hartnett, Intelligent Data Design, Inc. */
#include <mpi.h>
#include <stdio.h>

int main(int argc, char **argv)
{
    int rank, size;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    printf("Hello from rank %d of %d\n", rank, size);

    MPI_Finalize();
    return 0;
}
```

### 2. Create `examples/parallelIO/f_square16_par.f90`

```fortran
! f_square16_par.f90 - Parallel I/O hello: print MPI rank and size.
! Edward Hartnett, Intelligent Data Design, Inc.
program f_square16_par
  use mpi
  implicit none
  integer :: rank, size, ierr

  call MPI_Init(ierr)
  call MPI_Comm_rank(MPI_COMM_WORLD, rank, ierr)
  call MPI_Comm_size(MPI_COMM_WORLD, size, ierr)

  print *, 'Hello from rank', rank, 'of', size

  call MPI_Finalize(ierr)
end program f_square16_par
```

### 3. Create `examples/parallelIO/run_par_examples.sh.in`

Pattern from `/fast/ed/n1/nc_test4/run_par_test.sh.in`. The `@MPIEXEC@` token is substituted at configure time.

```sh
#!/bin/sh
# run_par_examples.sh.in - Run parallel I/O example programs.
# Processed at configure time: @MPIEXEC@ is replaced with the mpiexec path.
# Edward Hartnett, Intelligent Data Design, Inc.

set -e

echo ""
echo "Testing parallel I/O hello (C) with 4 processes..."
@MPIEXEC@ -n 4 ./square16_par

echo ""
echo "Testing parallel I/O hello (Fortran) with 4 processes..."
@MPIEXEC@ -n 4 ./f_square16_par

echo ""
echo "All parallel I/O example tests passed."
```

The script exits non-zero (via `set -e`) if either program returns a non-zero exit code.

### 4. Autotools: Create `examples/parallelIO/Makefile.am`

```makefile
# Makefile.am for parallel I/O examples
# Only built when --enable-parallel-tests is specified.

if HAVE_PARALLEL_TESTS

AUTOMAKE_OPTIONS = parallel-tests

AM_CPPFLAGS = -I$(top_srcdir)/include
AM_FCFLAGS = $(CPPFLAGS)

# C parallel example
check_PROGRAMS = square16_par f_square16_par

square16_par_SOURCES = square16_par.c
square16_par_LDADD = -lnetcdf

f_square16_par_SOURCES = f_square16_par.f90
f_square16_par_LDADD = -lnetcdff

# Configure the run script with @MPIEXEC@
TESTS = run_par_examples.sh

noinst_SCRIPTS = run_par_examples.sh

EXTRA_DIST = run_par_examples.sh.in

# Set LD_LIBRARY_PATH so test binaries can find HDF5/NetCDF shared libs at runtime
AM_TESTS_ENVIRONMENT = LD_LIBRARY_PATH="@MY_LD_LIBRARY_PATH@:$$LD_LIBRARY_PATH"

# Allow MPI to run as root in restricted environments (e.g., GitHub Actions)
if OPENMPI
AM_TESTS_ENVIRONMENT += OMPI_ALLOW_RUN_AS_ROOT=1 OMPI_ALLOW_RUN_AS_ROOT_CONFIRM=1
endif

CLEANFILES = *.nc run_par_examples.sh

endif # HAVE_PARALLEL_TESTS
```

**Notes on `run_par_examples.sh` generation**: Add `run_par_examples.sh` to `AC_CONFIG_FILES` in `configure.ac` so Autotools substitutes `@MPIEXEC@`. Mark the generated script executable with a `chmod` in the configure or via `check-local`.

Alternatively (simpler): generate via an `all-local` / `check-local` target:

```makefile

	chmod +x run_par_examples.sh square16_par f_square16_par
```

### 6. CMake: Create `examples/parallelIO/CMakeLists.txt`

```cmake
# CMakeLists.txt for parallel I/O examples
# Only built when ENABLE_PARALLEL_TESTS is ON (set by Sprint 2 build system work).

# Find MPI (already used for compilers; this picks up the include paths)
find_package(MPI REQUIRED)

# C parallel example
add_executable(square16_par square16_par.c)
target_include_directories(square16_par PRIVATE ${MPI_C_INCLUDE_PATH})
target_link_libraries(square16_par PRIVATE ${MPI_C_LIBRARIES})

# Fortran parallel example
add_executable(f_square16_par f_square16_par.f90)
target_include_directories(f_square16_par PRIVATE ${MPI_Fortran_INCLUDE_PATH})
target_link_libraries(f_square16_par PRIVATE ${MPI_Fortran_LIBRARIES})

# Configure the run script with MPIEXEC_EXECUTABLE
configure_file(run_par_examples.sh.in run_par_examples.sh @ONLY)

# Register the shell script as a CTest test
if(BUILD_TESTING)
    add_test(
        NAME parallel_examples
        COMMAND bash run_par_examples.sh
        WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
    )
    # Allow OpenMPI to run as root (GitHub Actions CI)
    set_tests_properties(parallel_examples PROPERTIES
        ENVIRONMENT "OMPI_ALLOW_RUN_AS_ROOT=1;OMPI_ALLOW_RUN_AS_ROOT_CONFIRM=1"
    )
endif()
```

**Note on `run_par_examples.sh.in`**: CMake's `configure_file` with `@ONLY` replaces `@MPIEXEC@` using the CMake variable `MPIEXEC` (found by `find_package(MPI)` or set via `MPIEXEC_EXECUTABLE` in Sprint 2). Confirm that `MPIEXEC` is set; if Sprint 2 uses `MPIEXEC_EXECUTABLE`, add `set(MPIEXEC ${MPIEXEC_EXECUTABLE})` before `configure_file`.

The workflow currently passes `-DBUILD_EXAMPLES=OFF` (CMake) and `--disable-examples` (Autotools). Sprint 3 needs examples enabled so `examples/parallelIO/` is built and tested.

**CMake job** — change the configure step:
```yaml
-DBUILD_EXAMPLES=ON \
```
(remove `-DBUILD_EXAMPLES=OFF`)

**Autotools job** — remove `--disable-examples` from configure flags:
```yaml
# Remove: --disable-examples
```

Also ensure the Autotools test step prints the `examples/parallelIO/test-suite.log` on failure:
```yaml
- name: Test (Autotools)
  if: matrix.build_system == 'autotools'
  run: |
    export LD_LIBRARY_PATH="..."
    make check ... \
      || (echo "Tests failed. Printing test logs:"; \
          cat test/test-suite.log 2>/dev/null; \
          cat examples/parallelIO/test-suite.log 2>/dev/null; \
          exit 1)
```

**OpenMPI root workaround**: GitHub Actions runners run as root. OpenMPI refuses to run as root by default. The `OMPI_ALLOW_RUN_AS_ROOT=1 OMPI_ALLOW_RUN_AS_ROOT_CONFIRM=1` environment variables must be set during the test step for the `openmpi` matrix job:

```yaml
- name: Test (CMake)
  if: matrix.build_system == 'cmake'
  working-directory: build
  run: |
    export LD_LIBRARY_PATH="..."
    if [ "${{ matrix.mpi }}" = "openmpi" ]; then
      export OMPI_ALLOW_RUN_AS_ROOT=1
      export OMPI_ALLOW_RUN_AS_ROOT_CONFIRM=1
    fi
    ctest --verbose
```

Apply the same pattern to the Autotools test step.

## Notes

- The programs make no NetCDF calls in Sprint 3. This isolates MPI infrastructure testing from NetCDF I/O complexity.
- `run_par_examples.sh.in` uses `set -e` so any non-zero exit from `mpiexec` fails the test immediately.
- OpenMPI on GitHub Actions requires `OMPI_ALLOW_RUN_AS_ROOT=1` because the CI runner runs as root. MPICH does not have this restriction.
- Sprint 2 must be implemented before Sprint 3 can be built — specifically `--enable-parallel-tests`, `MPIEXEC` path substitution in `configure.ac`, and `ENABLE_PARALLEL_TESTS` in `CMakeLists.txt`.
- The Fortran program uses `use mpi` (not `include 'mpif.h'`) and `.f90` free-form source, consistent with all other NEP Fortran examples.
- `f_square16_par` links against `-lnetcdff` in Autotools since the Fortran MPI module is typically bundled with the MPI installation but the Fortran compiler wrapper (`mpif90`) handles the link flags automatically; adjust if needed based on configure output.
