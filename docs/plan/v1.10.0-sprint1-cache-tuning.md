# V1.10.0 Sprint 1: Add example: cache_tuning.c — Chunk Cache Configuration

**Objective**: Create `examples/performance/cache_tuning.c` demonstrating `nc_set_chunk_cache()` and `nc_set_var_chunk_cache()` with measurable performance differences. The example shows default cache vs enlarged cache for repeated chunk access, file-level vs variable-level cache settings, and cache thrashing detection.

---

## Background

NetCDF-4/HDF5 uses a chunk cache to improve performance when accessing chunked data. The default cache size is often too small for optimal performance with large datasets or specific access patterns. This sprint creates a performance example that:

- Demonstrates the measurable impact of proper cache configuration
- Shows the difference between file-level and variable-level cache settings
- Detects cache thrashing when the access pattern exceeds available cache

**Dataset**: 3D temperature, 500×180×360, chunked 10×45×90 (~600KB/chunk)
**Access Pattern**: Reads same slabs repeatedly to demonstrate cache hit benefits

### 1. Create `examples/performance/cache_tuning.c`

**File**: `examples/performance/cache_tuning.c`

Create a comprehensive cache tuning example:

```c
/* cache_tuning.c - Demonstrate NetCDF-4 chunk cache configuration for performance.
 * Shows default vs enlarged cache, file-level vs variable-level settings,
 * and cache thrashing detection.
 * Edward Hartnett, Intelligent Data Design, Inc. */

#include <nc4_vars.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define NDIMS 3
#define NX 360
#define NY 180
#define NZ 500
#define CHUNK_X 90
#define CHUNK_Y 45
#define CHUNK_Z 10

/* Check NetCDF call result - standard NEP pattern */
#define NC_CHK(err) do { \
    if ((err) != NC_NOERR) { \
        fprintf(stderr, "Error at line %d: %s\n", __LINE__, nc_strerror(err)); \
        return 1; \
    } \
} while(0)

/* Get current time in seconds (for timing measurements) */
#include <sys/time.h>
double get_time(void)
{
    struct timeval tv;
    gettimeofday(&tv, NULL);
    return tv.tv_sec + tv.tv_usec / 1e6;
}

int main(int argc, char **argv)
{
    int ncid, varid, dimids[NDIMS];
    size_t chunksizes[NDIMS] = {CHUNK_Z, CHUNK_Y, CHUNK_X};
    size_t start[NDIMS], count[NDIMS];
    float *data;
    double t_start, t_end;
    int iter;
    size_t slab_size;
    size_t default_cache_size, default_nelems;
    float default_preemption;
    int i, j, k;
    int err;

    /* Create sample data file if it doesn't exist */
    err = nc_open("cache_test.nc", NC_NOWRITE, &ncid);
    if (err != NC_NOERR) {
        /* File doesn't exist, create it */
        NC_CHK(nc_create("cache_test.nc", NC_NETCDF4, &ncid));
        
        NC_CHK(nc_def_dim(ncid, "time", NZ, &dimids[0]));
        NC_CHK(nc_def_dim(ncid, "lat", NY, &dimids[1]));
        NC_CHK(nc_def_dim(ncid, "lon", NX, &dimids[2]));
        
        NC_CHK(nc_def_var(ncid, "temperature", NC_FLOAT, NDIMS, dimids, &varid));
        NC_CHK(nc_def_var_chunking(ncid, varid, NC_CHUNKED, chunksizes));
        
        NC_CHK(nc_enddef(ncid));
        
        /* Generate and write test data */
        slab_size = CHUNK_X * CHUNK_Y * CHUNK_Z;
        data = (float *)malloc(slab_size * sizeof(float));
        if (!data) {
            fprintf(stderr, "Memory allocation failed\n");
            return 1;
        }
        
        /* Write data chunk by chunk */
        for (k = 0; k < NZ; k += CHUNK_Z) {
            for (j = 0; j < NY; j += CHUNK_Y) {
                for (i = 0; i < NX; i += CHUNK_X) {
                    size_t idx;
                    for (idx = 0; idx < slab_size; idx++) {
                        data[idx] = 280.0f + (float)(k + idx % CHUNK_Z) * 0.1f;
                    }
                    start[0] = k; start[1] = j; start[2] = i;
                    count[0] = CHUNK_Z; count[1] = CHUNK_Y; count[2] = CHUNK_X;
                    if (k + CHUNK_Z > NZ) count[0] = NZ - k;
                    if (j + CHUNK_Y > NY) count[1] = NY - j;
                    if (i + CHUNK_X > NX) count[2] = NX - i;
                    NC_CHK(nc_put_vara_float(ncid, varid, start, count, data));
                }
            }
        }
        
        free(data);
        NC_CHK(nc_close(ncid));
        
        /* Reopen for reading tests */
        NC_CHK(nc_open("cache_test.nc", NC_NOWRITE, &ncid));
        NC_CHK(nc_inq_varid(ncid, "temperature", &varid));
    }

    /* Get default cache settings */
    NC_CHK(nc_get_var_chunk_cache(ncid, varid, &default_cache_size, 
                                   &default_nelems, &default_preemption));
    printf("Default cache settings:\n");
    printf("  cache_size: %zu bytes (%.1f MB)\n", 
           default_cache_size, default_cache_size / (1024.0 * 1024.0));
    printf("  nelems: %zu\n", default_nelems);
    printf("  preemption: %.2f\n\n", default_preemption);

    /* Allocate read buffer */
    slab_size = CHUNK_X * CHUNK_Y * CHUNK_Z;
    data = (float *)malloc(slab_size * sizeof(float));
    if (!data) {
        fprintf(stderr, "Memory allocation failed\n");
        return 1;
    }

    printf("Dataset: %dx%dx%d temperature, chunked %dx%dx%d (~%.0f KB/chunk)\n",
           NZ, NY, NX, CHUNK_Z, CHUNK_Y, CHUNK_X,
           (CHUNK_X * CHUNK_Y * CHUNK_Z * sizeof(float)) / 1024.0);
    printf("Access pattern: Reading same %zu chunks repeatedly\n\n", 
           (NZ / CHUNK_Z) * (NY / CHUNK_Y) * (NX / CHUNK_X));

    /* Test 1: Default cache - read all chunks 5 times */
    printf("Test 1: Default cache - 5 full passes through dataset\n");
    t_start = get_time();
    for (iter = 0; iter < 5; iter++) {
        for (k = 0; k < NZ; k += CHUNK_Z) {
            for (j = 0; j < NY; j += CHUNK_Y) {
                for (i = 0; i < NX; i += CHUNK_X) {
                    start[0] = k; start[1] = j; start[2] = i;
                    count[0] = CHUNK_Z; count[1] = CHUNK_Y; count[2] = CHUNK_X;
                    if (k + CHUNK_Z > NZ) count[0] = NZ - k;
                    if (j + CHUNK_Y > NY) count[1] = NY - j;
                    if (i + CHUNK_X > NX) count[2] = NX - i;
                    NC_CHK(nc_get_vara_float(ncid, varid, start, count, data));
                }
            }
        }
    }
    t_end = get_time();
    printf("  Time: %.3f seconds\n\n", t_end - t_start);

    /* Test 2: Range of cache sizes to show performance scaling */
    printf("Test 2: Cache size scaling - testing 4MB, 16MB, 64MB, 128MB\n");
    
    size_t cache_sizes[] = {4*1024*1024, 16*1024*1024, 64*1024*1024, 128*1024*1024};
    const char *cache_labels[] = {"4 MB", "16 MB", "64 MB", "128 MB"};
    double default_time = t_end - t_start;
    
    for (int c = 0; c < 4; c++) {
        /* Set cache size */
        NC_CHK(nc_set_var_chunk_cache(ncid, varid, cache_sizes[c], 1009, 0.75));
        
        /* Warm up cache */
        for (k = 0; k < NZ; k += CHUNK_Z) {
            for (j = 0; j < NY; j += CHUNK_Y) {
                for (i = 0; i < NX; i += CHUNK_X) {
                    start[0] = k; start[1] = j; start[2] = i;
                    count[0] = CHUNK_Z; count[1] = CHUNK_Y; count[2] = CHUNK_X;
                    if (k + CHUNK_Z > NZ) count[0] = NZ - k;
                    if (j + CHUNK_Y > NY) count[1] = NY - j;
                    if (i + CHUNK_X > NX) count[2] = NX - i;
                    NC_CHK(nc_get_vara_float(ncid, varid, start, count, data));
                }
            }
        }
        
        /* Timed run */
        t_start = get_time();
        for (iter = 0; iter < 5; iter++) {
            for (k = 0; k < NZ; k += CHUNK_Z) {
                for (j = 0; j < NY; j += CHUNK_Y) {
                    for (i = 0; i < NX; i += CHUNK_X) {
                        start[0] = k; start[1] = j; start[2] = i;
                        count[0] = CHUNK_Z; count[1] = CHUNK_Y; count[2] = CHUNK_X;
                        if (k + CHUNK_Z > NZ) count[0] = NZ - k;
                        if (j + CHUNK_Y > NY) count[1] = NY - j;
                        if (i + CHUNK_X > NX) count[2] = NX - i;
                        NC_CHK(nc_get_vara_float(ncid, varid, start, count, data));
                    }
                }
            }
        }
        t_end = get_time();
        
        double elapsed = t_end - t_start;
        printf("  %s: %.3f seconds (%.1fx vs default)\n", 
               cache_labels[c], elapsed, default_time / elapsed);
    }
    printf("\n");

    /* Restore default cache for comparison */
    NC_CHK(nc_set_var_chunk_cache(ncid, varid, default_cache_size, 
                                   default_nelems, default_preemption));

    /* Test 3: Cache thrashing detection */
    printf("Test 3: Cache thrashing detection\n");
    printf("  Reading same small slab repeatedly with small cache...\n");
    
    /* Set very small cache to force thrashing */
    NC_CHK(nc_set_var_chunk_cache(ncid, varid, 4 * 1024 * 1024, 1009, 0.75));
    
    /* Read 10 chunks repeatedly 100 times - if cache < 10 chunks, thrashing occurs */
    int thrash_chunks = 10;
    t_start = get_time();
    for (iter = 0; iter < 100; iter++) {
        for (k = 0; k < thrash_chunks * CHUNK_Z && k < NZ; k += CHUNK_Z) {
            start[0] = k; start[1] = 0; start[2] = 0;
            count[0] = CHUNK_Z; count[1] = CHUNK_Y; count[2] = CHUNK_X;
            NC_CHK(nc_get_vara_float(ncid, varid, start, count, data));
        }
    }
    t_end = get_time();
    printf("  Time with small cache (4 MB, fits ~%d chunks): %.3f seconds\n",
           (4 * 1024 * 1024) / (CHUNK_X * CHUNK_Y * CHUNK_Z * sizeof(float)),
           t_end - t_start);
    
    /* Now enlarge cache to fit all thrash_chunks */
    NC_CHK(nc_set_var_chunk_cache(ncid, varid, 
                                   thrash_chunks * CHUNK_X * CHUNK_Y * CHUNK_Z * sizeof(float) * 2,
                                   1009, 0.75));
    
    t_start = get_time();
    for (iter = 0; iter < 100; iter++) {
        for (k = 0; k < thrash_chunks * CHUNK_Z && k < NZ; k += CHUNK_Z) {
            start[0] = k; start[1] = 0; start[2] = 0;
            count[0] = CHUNK_Z; count[1] = CHUNK_Y; count[2] = CHUNK_X;
            NC_CHK(nc_get_vara_float(ncid, varid, start, count, data));
        }
    }
    t_end = get_time();
    printf("  Time with enlarged cache (fits %d chunks): %.3f seconds\n",
           thrash_chunks * 2, t_end - t_start);
    printf("  Thrashing resolved: cache now holds all working set chunks\n\n");

    /* Test 4: File-level cache settings and additional API coverage */
    printf("Test 4: File-level cache via nc_set_chunk_cache() and nc_get_chunk_cache()\n");
    
    /* Close and reopen to test file-level settings */
    NC_CHK(nc_close(ncid));
    
    /* Get current file-level cache defaults */
    size_t file_cache_size, file_nelems;
    float file_preemption;
    NC_CHK(nc_get_chunk_cache(&file_cache_size, &file_nelems, &file_preemption));
    printf("  Default file-level cache: %.1f MB, nelems=%zu, preemption=%.2f\n",
           file_cache_size / (1024.0 * 1024.0), file_nelems, file_preemption);
    
    /* Set file-level cache before opening */
    NC_CHK(nc_set_chunk_cache(64 * 1024 * 1024, 1009, 0.75));
    
    /* Verify the setting took effect */
    NC_CHK(nc_get_chunk_cache(&file_cache_size, &file_nelems, &file_preemption));
    printf("  After nc_set_chunk_cache(): %.1f MB, nelems=%zu, preemption=%.2f\n",
           file_cache_size / (1024.0 * 1024.0), file_nelems, file_preemption);
    
    NC_CHK(nc_open("cache_test.nc", NC_NOWRITE, &ncid));
    NC_CHK(nc_inq_varid(ncid, "temperature", &varid));
    
    /* Check chunking info via nc_inq_var_chunking() */
    int storage;
    size_t var_chunks[NDIMS];
    NC_CHK(nc_inq_var_chunking(ncid, varid, &storage, var_chunks));
    printf("  Variable storage: %s, chunks: %zux%zux%zu\n",
           (storage == NC_CHUNKED) ? "chunked" : "contiguous",
           var_chunks[0], var_chunks[1], var_chunks[2]);
    
    /* Check that new file has enlarged cache */
    size_t var_cache_size;
    NC_CHK(nc_get_var_chunk_cache(ncid, varid, &var_cache_size, NULL, NULL));
    printf("  File opened with variable cache_size: %.1f MB\n\n", 
           var_cache_size / (1024.0 * 1024.0));
    
    /* Cleanup */
    free(data);
    NC_CHK(nc_close(ncid));
    
    /* Remove test file */
    remove("cache_test.nc");
    
    printf("All cache tuning tests complete.\n");
    printf("\nKey takeaways:\n");
    printf("  - Default cache may be too small for large datasets\n");
    printf("  - Variable-level cache: nc_set_var_chunk_cache() for specific variables\n");
    printf("  - File-level cache: nc_set_chunk_cache() affects new file opens\n");
    printf("  - Cache thrashing occurs when working set exceeds cache size\n");
    printf("  - Proper tuning can provide significant performance improvement\n");
    
    return 0;
}
```
